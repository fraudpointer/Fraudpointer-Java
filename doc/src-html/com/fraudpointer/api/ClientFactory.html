001    package com.fraudpointer.api;
002    
003    import com.fraudpointer.api.clients.Client;
004    
005    /**
006     * This is a helping class that instantiates an object that implements IClient interface. <br/>
007     * <br/>
008     * Use this method to start with FraudPointer API. It will give you an object that properly implements IClient interface.
009     */
010    public class ClientFactory {
011    
012        /**
013         * Instantiates an object that properly implements IClient interface. <br/>
014         * <br/>
015         * Start your FraudPointer API usage from here. It will give you an instance of an object that properly implements the IClient interface.<br/>
016         * <br/>
017         * Here is an example of usage:<br/>
018         * <br/>
019         * <pre>
020         * IClient client = ClientFactory.construct("https://production.fraudpointer.com/api", "111118312453678901abcdef123456789328");
021         * </pre>
022         * <br/>
023         * @param baseUrl This is the URL of the FraudPointer API Service. It has to have the value: <code>https://production.fraudpointer.com/api</code>
024         * @param apiKey This should be the API KEY that corresponds to the domain that you integrate FraudPointer API for. The API KEY is automatically generated when
025         * you register an Account in <a href="https://production.fraudpointer.com/a/new" target="_blank">FraudPointer Registration Web Form</a>.
026         * Note that your Account in FraudPointer service might have more than one domains registered. You need to provide here the API KEY that
027         * corresponds to the domain that you are going to integrate this client with.
028         * @return A valid IClient instantiated object. This is a newly created object to work with the FraudPointer Service.
029         */
030        public static IClient construct (String baseUrl, String apiKey)
031        {
032            return new Client(baseUrl, apiKey);
033    
034        } // construct()
035        // -------------
036    
037        private ClientFactory(){}
038    
039    } // class ClientFactory
040    //-----------------------
