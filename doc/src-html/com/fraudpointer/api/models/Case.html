001    package com.fraudpointer.api.models;
002    
003    import com.google.gson.annotations.SerializedName;
004    import java.util.Date;
005    
006    /**
007     * A Case is created for every <b>final</b> {@link com.fraudpointer.api.models.FraudAssessment FraudAssessment}. <br/>
008     * <br/>
009     * The Case is meant to be handled by a human interacting with the Fraudpointe Application. When a Case is created, it has
010     * an "Open" status (unless you have rules in your Fraudpointer application configuration that automatically put them in other statuses).
011     * When a Case is inspected by a human, human might decide to set a {@link com.fraudpointer.api.models.Case#resolution resolution} to it.
012     */
013    public class Case {
014    
015        /**
016         * The Resolution of a Case as set by human (or automatically according to Fraudpointer application configuration). <br/>
017         * <br/>
018         * Can take the following values:<br/>
019         * <br/>
020         * <ul>
021         *     <li>"Accept"</li>
022         *     <li>"Review"</li>
023         *     <li>"Reject"</li>
024         * </ul>
025         */
026        public String resolution;
027    
028        /**
029         * Case Id for further reference
030         */
031        public String id;
032    
033        /**
034         * Case status. <br/>
035         * <br/>
036         * Can take the following values:<br/>
037         * <br/>
038         * <ul>
039         *     <li>"Open"</li>
040         *     <li>"Closed"</li>
041         * </ul>
042         *
043         */
044        public String status;
045    
046        /**
047         * Last time the Case was updated
048         */
049        @SerializedName("updated_at") public Date updatedAt;
050    
051    } // class Case
