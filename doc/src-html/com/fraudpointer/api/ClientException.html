003    /**
004     * Thrown by all IClient methods in case an error takes place and method cannot return its results normally. <br/>
005     * <br/>
006     * You should always put your IClient methods into <code>try/catch</code> blocks and make sure you catch this exception.
007     * If you catch this exception, the message borne will be adequate to explain to you what has gone wrong.<br/>
008     * <br/>
009     * Here is how you can wrap your code (example of a Console Application) around <code>try/catch</code> block
010     * and on error get information about what has gone wrong:<br/>
011     *
012     * <pre>
013     * try {
014     *     ... your Fraudpointer code goes here ....
015     * }
016     * catch (Exception ex)
017     * {
018     *   System.err.println(ex.getMessage());
019     *   ex.printStackTrace();
020     * }
021     * </pre>
022     */
023    public class ClientException extends Exception {
024        public ClientException (String message, Exception innerException)
025        {
026            super(message, innerException);
027        }
028    } // class ClientException
