001    package com.fraudpointer;
002    
003    import com.fraudpointer.api.ClientException;
004    import com.fraudpointer.api.ClientFactory;
005    import com.fraudpointer.api.IClient;
006    
007    /**
008     * This is a console application that can be used to encrypt a credit card hash the way Fraudpointer service requires you to do. <br/>
009     * <br/>
010     * We emphasize here that when you want to send the credit card to Fraudpointer Server, you <b>should not send it in plain clear format</b>.
011     * You <b>should use the {@link com.fraudpointer.api.IClient#creditCardHash(String) creditCardHash} method</b>, like this sample application does
012     * below.<br/>
013     * <br/>
014     * Note that you can use <code>java -cp fraudpointer_client.jar com.fraudpointer.EncryptCreditCard &lt;credit_card_to_encrypt&gt;</code> if you want to
015     * immediately run this utility.
016     */
017    public class EncryptCreditCard {
018        public static void main (String[] args)
019        {
020            if (args.length != 1) {
021                wrongSyntax();
022                return;
023            }
024    
025            String creditCard = args[0];
026    
027            IClient fraudpointerClient = ClientFactory.construct("", "");
028            String encryptedCreditCard = "";
029            try {
030                encryptedCreditCard = fraudpointerClient.creditCardHash(creditCard);
031            }
032            catch (ClientException ex) {
033                ex.printStackTrace();
034                return;
035            }
036    
037            System.out.println("Encrypted credit card: " + encryptedCreditCard);
038    
039    
040        } // main ()
041        //----------
042    
043        protected static void wrongSyntax ()
044        {
045            System.err.println("Correct syntax is: com.fraudpointer.EncryptCreditCard <credit_card_number>");
046        } // wrongSyntax()
047        //-----------------
048    
049    } // class EncryptCreditCard
050    //---------------------------
